/*Section with video attribute*/
"use client";

import { useRef, useState } from "react";
import Link from "next/link";
import { ArrowRight, Play, Pause } from "lucide-react";

export default function HeroSection() {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [isPlaying, setIsPlaying] = useState(true);

  const togglePlayPause = () => {
    if (!videoRef.current) return;
    if (isPlaying) {
      videoRef.current.pause();
    } else {
      videoRef.current.play();
    }
    setIsPlaying(!isPlaying);
  };

  return (
    <section className="w-full min-h-screen flex flex-col md:flex-row">
      {/* Left Column */}
      <div className="bg-gray-900 text-amber-50 flex flex-col justify-center px-8 py-16 md:w-1/2">
        <h1 className="text-4xl md:text-5xl font-bold leading-tight mb-6">
          Advanced Earth Observation & Analysis
        </h1>
        <p className="text-lg text-amber-100 mb-8">
          Harness the power of satellite imagery and geospatial intelligence to monitor our planet, analyze environmental change, and drive informed climate action.
        </p>
        <Link
          href="/arcgis"
          className="inline-flex items-center bg-amber-50 hover:bg-amber-100 text-gray-900 px-6 py-3 rounded-lg font-semibold transition"
        >
          Learn about Us
          <ArrowRight className="ml-2 w-5 h-5" />
        </Link>
      </div>

      {/* Right Column */}
      <div className="relative md:w-1/2 h-64 md:h-auto">
        {/* Video Background */}
        <video
          ref={videoRef}
          src="/videos/vid_4.mp4"
          autoPlay
          loop
          muted
          playsInline
          className="w-full h-full object-cover"
        />

        {/* Play/Pause Button */}
        <button
          onClick={togglePlayPause}
          className="absolute top-4 right-4 bg-gray-900 bg-opacity-60 hover:bg-opacity-80 text-amber-50 p-2 rounded-full transition"
        >
          {isPlaying ? <Pause className="w-5 h-5" /> : <Play className="w-5 h-5" />}
        </button>
      </div>
    </section>
  );
}
